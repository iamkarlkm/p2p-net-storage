        writeInt(bytes, index, inode.i_sys_names_offset);
        index += INT_SIZE;

        // 写入 extra_data
        for (long data : inode.extra_data) {
            writeLong(bytes, index, data);
            index += LONG_SIZE;
        }

        // 写入 sn
        writeLong(bytes, index, inode.sn);
        index += LONG_SIZE;

        // 写入 inode_ctime
        writeLong(bytes, index, inode.inode_ctime);
        index += LONG_SIZE;

        // 写入 i_ext_super_mount_nodes
        writeLong(bytes, index, inode.i_ext_super_mount_nodes);
        index += LONG_SIZE;

        // 写入 i_ext_super_nodes
        writeLong(bytes, index, inode.i_ext_super_nodes);
        index += LONG_SIZE;

        // 写入 md5
        System.arraycopy(inode.md5, 0, bytes, index, MD5_SIZE);
        index += MD5_SIZE;

        // 写入 i_next_mvcc_super_node
        writeLong(bytes, index, inode.i_next_mvcc_super_node);
        index += LONG_SIZE;

        // 写入 i_exec_entry_node
        writeLong(bytes, index, inode.i_exec_entry_node);

        return bytes;
    }

    public static Ds128SuperInode fromByteArray(byte[] bytes) {
        Ds128SuperInode inode = new Ds128SuperInode();
        int index = 0;

        // 读取 magic
        inode.magic = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 i_root_node
        inode.i_root_node = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 mft_size
        inode.mft_size = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 block_total
        inode.block_total = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 block_size
        inode.block_size = readInt(bytes, index);
        index += INT_SIZE;

        // 读取 i_sys_names_offset
        inode.i_sys_names_offset = readInt(bytes, index);
        index += INT_SIZE;

        // 读取 extra_data
        for (int i = 0; i < inode.extra_data.length; i++) {
            inode.extra_data[i] = readLong(bytes, index);
            index += LONG_SIZE;
        }

        // 读取 sn
        inode.sn = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 inode_ctime
        inode.inode_ctime = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 i_ext_super_mount_nodes
        inode.i_ext_super_mount_nodes = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 i_ext_super_nodes
        inode.i_ext_super_nodes = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 md5
        System.arraycopy(bytes, index, inode.md5, 0, MD5_SIZE);
        index += MD5_SIZE;

        // 读取 i_next_mvcc_super_node
        inode.i_next_mvcc_super_node = readLong(bytes, index);
        index += LONG_SIZE;

        // 读取 i_exec_entry_node
        inode.i_exec_entry_node = readLong(bytes, index);

        return inode;
    }

    private static void writeLong(byte[] bytes, int index, long value) {
        for (int i = 0; i < LONG_SIZE; i++) {
            bytes[index + i] = (byte) (value >> (i * 8));
        }
    }

    private static long readLong(byte[] bytes, int index) {
        long value = 0;
        for (int i = 0; i < LONG_SIZE; i++) {
            value |= ((long) (bytes[index + i] & 0xFF)) << (i * 8);
        }
        return value;
    }

    private static void writeInt(byte[] bytes, int index, int value) {
        for (int i = 0; i < INT_SIZE; i++) {
            bytes[index + i] = (byte) (value >> (i * 8));
        }
    }

    private static int readInt(byte[] bytes, int index) {
        int value = 0;
        for (int i = 0; i < INT_SIZE; i++) {
            value |= ((bytes[index + i] & 0xFF)) << (i * 8);
        }
        return value;
    }
}

在 Java 中，你可以使用 java.nio 包中的 FileChannel 和 MappedByteBuffer 来实现内存映射文件的读写操作。下面是一个示例代码，展示了如何使用内存映射方式读写 Ds128SuperInode 对象：
import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.RandomAccessFile;
import java.nio.MappedByteBuffer;
import java.nio.channels.FileChannel;

class Ds128SuperInode {
    // 8字节，文件系统标志和版本号，当前为DS64v1.0
    //long magic;
	byte[] magic = "DS64v1.0".getBytes();
    // 根节点
    long i_root_node;
    // Size in bytes
    long mft_size;
    // 块占用总数
    long block_total;
    // 块大小
    int block_size;
    // 系统内部命名标识符字符串位置偏移
    int i_sys_names_offset;
    // 短命名标识符存储
    long[] extra_data = new long[3];
    // 时间戳序列号
    long sn;
    // inode创建时间
    long inode_ctime;
    // 文件模式存储空间扩展
    long i_ext_super_mount_nodes;
    // 块设备模式存储空间扩展
    long i_ext_super_nodes;
    byte[] md5 = new byte[16];
    // 多版本并发控制 -- 文件系统
    long i_next_mvcc_super_node;
    // 可执行代码入口 main entry,boot entry
    long i_exec_entry_node;

    // 写入到内存映射缓冲区
    public void writeToMappedByteBuffer(MappedByteBuffer buffer) {
        buffer.putLong(magic);
        buffer.putLong(i_root_node);
        buffer.putLong(mft_size);
        buffer.putLong(block_total);
        buffer.putInt(block_size);
        buffer.putInt(i_sys_names_offset);
        for (long data : extra_data) {
            buffer.putLong(data);
        }
        buffer.putLong(sn);
        buffer.putLong(inode_ctime);
        buffer.putLong(i_ext_super_mount_nodes);
        buffer.putLong(i_ext_super_nodes);
        buffer.put(md5);
        buffer.putLong(i_next_mvcc_super_node);
        buffer.putLong(i_exec_entry_node);
    }

    // 从内存映射缓冲区读取
    public void readFromMappedByteBuffer(MappedByteBuffer buffer) {
        magic = buffer.getLong();
        i_root_node = buffer.getLong();
        mft_size = buffer.getLong();
        block_total = buffer.getLong();
        block_size = buffer.getInt();
        i_sys_names_offset = buffer.getInt();
        for (int i = 0; i < extra_data.length; i++) {
            extra_data[i] = buffer.getLong();
        }
        sn = buffer.getLong();
        inode_ctime = buffer.getLong();
        i_ext_super_mount_nodes = buffer.getLong();
        i_ext_super_nodes = buffer.getLong();
        buffer.get(md5);
        i_next_mvcc_super_node = buffer.getLong();
        i_exec_entry_node = buffer.getLong();
    }
}

public class Ds128SuperInodeMemoryMappedIO {
    private static final int SIZE = 8 * 10 + 4 * 2 + 16; // 计算 Ds128SuperInode 的字节大小

    public static void writeToFile(Ds128SuperInode inode, String filePath) throws IOException {
        try (RandomAccessFile file = new RandomAccessFile(new File(filePath), "rw");
             FileChannel channel = file.getChannel()) {
            MappedByteBuffer buffer = channel.map(FileChannel.MapMode.READ_WRITE, 0, SIZE);
            inode.writeToMappedByteBuffer(buffer);
        }
    }

    public static Ds128SuperInode readFromFile(String filePath) throws IOException {
        try (RandomAccessFile file = new RandomAccessFile(new File(filePath), "r");
             FileChannel channel = file.getChannel()) {
            MappedByteBuffer buffer = channel.map(FileChannel.MapMode.READ_ONLY, 0, SIZE);
            Ds128SuperInode inode = new Ds128SuperInode();
            inode.readFromMappedByteBuffer(buffer);
            return inode;
        }
    }

    public static void main(String[] args) {
        Ds128SuperInode inode = new Ds128SuperInode();
        inode.magic = 0x123456789ABCDEFL;
        inode.i_root_node = 1L;
        // 其他字段赋值...

        String filePath = "super_inode.dat";
        try {
            writeToFile(inode, filePath);
            Ds128SuperInode readInode = readFromFile(filePath);
            System.out.println("Magic: " + readInode.magic);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

### 代码说明：
1. Ds128SuperInode 类 ：
   
   - 定义了 Ds128SuperInode 的数据结构。
   - writeToMappedByteBuffer 方法：将 Ds128SuperInode 对象的各个字段写入 MappedByteBuffer 。
   - readFromMappedByteBuffer 方法：从 MappedByteBuffer 中读取数据并填充到 Ds128SuperInode 对象的各个字段。
2. Ds128SuperInodeMemoryMappedIO 类 ：
   
   - SIZE 常量：计算 Ds128SuperInode 对象所需的字节大小。
   - writeToFile 方法：使用 FileChannel 和 MappedByteBuffer 将 Ds128SuperInode 对象写入文件。
   - readFromFile 方法：使用 FileChannel 和 MappedByteBuffer 从文件中读取数据并创建 Ds128SuperInode 对象。
   - main 方法：创建一个 Ds128SuperInode 对象，将其写入文件，然后从文件中读取并打印其中一个字段的值。
### 注意事项：
- 内存映射文件适用于处理大文件，因为它可以将文件的一部分映射到内存中，避免了频繁的磁盘 I/O 操作。
- 确保文件路径存在且有读写权限，否则会抛出 IOException 。
- 内存映射文件的大小必须在创建时指定，因此需要提前计算 Ds128SuperInode 对象所需的字节大小。
